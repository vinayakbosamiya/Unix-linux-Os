#Write a script that generates a secure random password.
#!/bin/bash
# Function to generate a random password
generate_password() 
{
tr -dc 'A-Za-z0-9!@#$%^&*()_+{}[]' < /dev/urandom | fold -w 12 | head -n 1
}
# Call the function and store the generated password
password=$(generate_password)
echo "Generated password: $password"

Create a script to add, modify, or delete user accounts on a Linux system.
#!/bin/bash
action="$1"
username="$2"
case $action in
    "add")
        useradd "$username"
        ;;
    "modify")
        usermod -s /bin/bash "$username"
        ;;
    "delete")
        userdel "$username"
        ;;
    *)
        echo "Usage: $0 {add|modify|delete} username"
        exit 1
        ;;
esac
